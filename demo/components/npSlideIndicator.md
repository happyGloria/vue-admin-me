# 标签名
np-slide-indicator

# Description :
横向滑动指示器

# 浏览器兼容性 :
Chrome，Firefox 

# Author :
葛伟

# Created :
2017-5-5

# API :

<pre>
props : [
	disabled : true, // default : false,
	min : Int, // 最小, default : 0,
	max : Int, // 最大, default : 100
	value: Int
]
events : [
	slide : 滑动过程中的value变化事件
	input : 滑动完成后的value改变事件
]
</pre>
